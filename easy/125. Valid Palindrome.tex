给定一个字符串，验证它是否是回文串，只考虑字母和数字字符，可以忽略字母的大小写。

说明：本题中，我们将空字符串定义为有效的回文串。

示例 1:

输入: "A man, a plan, a canal: Panama"
输出: true

示例 2:

输入: "race a car"
输出: false


























思路：
直接法。































code：
class Solution {
public:
    bool isPalindrome(string s) {
        vector<char> sv;
        for(int i = 0; i < s.size(); i++)
        {
            if(s[i] >= 'A' && s[i] <= 'Z')
                sv.push_back(s[i] + 32);
            else if(s[i] >= 'a' && s[i] <= 'z')
                sv.push_back(s[i]);
            else if(isdigit(s[i]))
                sv.push_back(s[i]);
            else continue;
        }
        for(int j = 0; j < sv.size()/2; j++)
        {
            if(sv[j] != sv[sv.size()-1-j]) return false;
        }
        return true;
    }
};
